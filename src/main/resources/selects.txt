SELECT
    address.address,
    address.num,
    address.parish,
    address.town,
    address.city,
    address.county,
    address.state_province,
    address.island,
    address.region,
    address.country,
    address.postal_code,
    email.email,
    phone.country_code,
    phone.number
FROM contact
INNER JOIN address ON contact.address_id = address.id
INNER JOIN email ON contact.email_id = email.id
INNER JOIN phone ON contact.phone_id = phone.id
